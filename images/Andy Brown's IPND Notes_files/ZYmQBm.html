@function join-selector($selector, $n) {
  $s: $selector;
  @if $n > 1 {
    @for $i from 1 to $n {
      $s: $s + "+" + $selector;
    }
  }
  @return unquote($s);
}

@mixin cut($angle, $old-depth, $width, $side) {
  $t: AAS($angle, 90deg, $height);
  $depth: $old-depth - map-get($t, $side) * 2 !global;
  $circ: $depth * pi();
  $new-width: $circ / $slices !global;
  $delta: ($width - $new-width) / 2;
  width: $width;
  margin-left: $width / -2;
  clip-path: polygon($delta 0, 0 $height, $width $height, $width - $delta 0, $delta 0);
}
